316 void DecisionTree::fit_(TreeNode* node)
317 {
318  DataFrame dataframe = node->getDataFrame();
319  LabelCounter label_counter = LabelCounter(dataframe.col(-1));
320  double proportion = label_counter.get_values().max()/label_counter.size();
321  if ( label_counter.size()==1 ) {
322  return; // Prune if there is only one class left.
323  } else if ( dataframe.length()<2 ) {
324  return; // Prune if there is not enough data to split.
325  } else if ( (this->max_height_!=-1) and (node->getDepth()+1>=this->max_height_) ) {
326  return; // Prune if adding children would exceed max depth:
327  } else if ( (this->max_leaves_!=-1) and (this->num_leaves_+1>=this->max_leaves_) ) {
328  return; // Prune if adding children would exceed max leaves:
329  } else if ( (this->min_obs_!=-1) and (dataframe.length()<=this->min_obs_) ) {
330  return; // Prune if node is below minimum leave size.
331  } else if ( (this->max_prop_!=-1) and ( proportion>=this->max_prop_) ) {
332  return; // Prune if proportion of majority label is above threshold.
333  }
334  // Find best split at this node:
335  std::pair<int,double> split = this->findBestSplit(node);
336  int split_feature = split.first;
337  double split_threshold = split.second;
338  // To handle scenario where all columns within mtry have just 1 unique value
339  if (split_feature == -1 && split_threshold == -1.0)
340  {
341  return;
342  }
343  node->setSplitFeature(split_feature);
344  node->setSplitThreshold(split_threshold);
345  // Calculate results of best split:
346  std::vector<DataFrame> dataset_splits = dataframe.split(split_feature,split_threshold,true); // equal_goes_left=true.
347  DataFrame left_data = dataset_splits[0];
348  DataFrame right_data = dataset_splits[1];
349  if ( (left_data.length()==0) or (right_data.length()==0) ) {
350  return; // Prune if best split does not actually split the dataset.
351  }
352  // If split produces two non-empty dataframes, recurse to (new) children:
353  this->num_leaves_ += 1; // Each split causes net addition of 1 leaf.
354  TreeNode *left_child = new TreeNode(left_data);
355  TreeNode *right_child = new TreeNode(right_data);
356  node->setLeft(left_child);
357  node->setRight(right_child);
358  // Recurse to (new) children:
359  this->fit_(left_child);
360  this->fit_(right_child);
361 }

363 double DecisionTree::predict_(DataVector* observation) const
364 {
366  TreeNode* node = this->root_;
367  // Starting at root node, traverse tree while going left or right according to trained splitting criteria:
368  while (!node->isLeaf())
369  {
370  int split_feature = node->getSplitFeature();
371  double split_threshold = node->getSplitThreshold();
372  if ( observation->value(split_feature) <= split_threshold ){
373  assert (node->hasLeft()); // A non-leaf node should have both left and right children.
374  node = node->getLeft();
375  } else {
376  assert (node->hasRight()); // A non-leaf node should have both left and right children.
377  node = node->getRight();
378  }
379  }
380  // Make prediction based on whichever leaf is reached by the traversal:
381  double prediction;
382  if (this->isRegressionTree()) {
383  // Regression tree: Predict mean of training data at terminal node:
384  prediction = node->getDataFrame().col(-1).mean();
385  } else {
386  // Classification tree: Predict majority class of training data at terminal node:
387  prediction = LabelCounter(node->getDataFrame().col(-1)).get_most_frequent();
388  }
389  return prediction;
