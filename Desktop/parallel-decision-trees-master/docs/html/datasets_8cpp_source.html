549 DataFrame DataFrame::sample(int nrow, int seed, bool replace) const{
550  // Set random seed for reproducibility if specified
551  if (seed == -1){
552  // obtain a random number from hardware
553  std::random_device rd;
554  seed = rd();
555  }
556  // number of rows to pull
557  if (nrow == -1){
558  // if default, bootstrap a sample of equal length
559  nrow = this->length();
560  } else if (replace == false){
561  // if sample without replacing, cap at original length
562  nrow = std::min(nrow, this->length());
563  }else{
564  assert(nrow > 0);
565  }
566  // Create new empty DataFrame
567  DataFrame new_frame = DataFrame();
568  if (replace == true){
569  // Seed the generator
570  std::mt19937 eng(seed);
571  // Draw row indices from uniform distribution
572  std::uniform_int_distribution<> distr(0, this->length()-1);
573  // pull random rows (as pointers, not copies) with replacement until full
574  while (new_frame.length() < nrow){
575  // get random row index with replacement
576  int rand_row = distr(eng);
577  // get pointer to that row in original dataframe and store in bootstrap
578  new_frame.addRow(this->row(rand_row));
579  }
580  }else{
581  // pre-allocate vector of row indices
582  std::vector<int> indices(this->length());
583  // fill vector, equivalent to np.arange(0, this->length()-1)
584  std::generate(indices.begin(), indices.end(), [n = 0] () mutable {return n++;});
585  // Seed random generator
586  srand((unsigned) seed);
587  // Shuffle vector of original df row indices
588  for (int i = 0; i < this->length(); i++){
589  std::swap(indices[i], indices[i+(std::rand() % (this->length()-i))]);
590  }
591  // pull random rows (as pointers, not copies) until full
